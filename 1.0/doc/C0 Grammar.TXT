<program> ::= <block>

<block> ::= "{" <stmt_list> "}"

<stmt_list> ::= <stmt_list> <stmt> 
              | ε

<stmt> ::= <assign_stmt>
         | <input_stmt> 
         | <output_stmt>

<assign_stmt> ::= <location> "=" <expr> ";"

<input_stmt> ::= "input" "(" <location> ")" ";"

<output_stmt> ::= "output" "(" <expr> ")" ";"

<expr> ::= <expr> <bi_op_1> <expr_1> 
         | <expr_1>

<expr_1> ::= <expr_1> <bi_op_2> <expr_2>
           | <expr_2>

<expr_2> ::= <location>
           | <literal>

<bi_op_1> ::= "+"

<bi_op_2> ::= "*"

<location> ::= id

<literal> ::= integer_literal
